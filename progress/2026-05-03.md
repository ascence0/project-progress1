'''
#define POT_PIN 34

unsigned long lastBeat = 0;
float bpm = 0;

bool lastState = false;  // 上一次状态

void setup() {
  Serial.begin(115200);
}

void loop() {
  int potValue = analogRead(POT_PIN);

  float simulatedHR = map(potValue, 0, 4095, 60, 120);
  float freq = simulatedHR / 60.0;

  static float t = 0;

  float signal = sin(2 * PI * freq * t);
  signal += random(-10, 10) / 50.0;

  int value = (signal + 1) * 2048;

  // ===== 核心修正 =====
  bool currentState = (value > 3000);

  // 只检测“上升沿”
  if (currentState && !lastState) {
    unsigned long now = millis();

    // 冷却时间（防止误触发）
    if (now - lastBeat > 400) {
      if (lastBeat != 0) {
        float interval = (now - lastBeat) / 1000.0;
        bpm = 60.0 / interval;

        Serial.print("SimHR:");
        Serial.print(simulatedHR);
        Serial.print(" | BPM:");
        Serial.println(bpm);
      }

      lastBeat = now;
    }
  }

  lastState = currentState;

  t += 0.02;
  delay(20);
}
'''
